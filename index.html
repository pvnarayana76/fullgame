const currentWordData = { fullWord: "केला", parts: ["क", "े", "ल", "ा"], // We can leave these empty or add "distractors" here extraVyanjan: ["न", "म", "प", "र"], extraMatra: ["ि", "ु", "ो"] }; function initGame() { const vMixer = document.getElementById('vyanjan-mixer'); const mMixer = document.getElementById('matra-mixer'); // 1. Identify which parts are Vyanjans and which are Matras // (In Hindi, Matras usually have distinct Unicode ranges or we can list them) const matraList = ["ा", "ि", "ी", "ु", "ू", "े", "ै", "ो", "ौ", "ं", "ः", "ृ"]; // 2. Create the Vyanjan Pool (Required + Extras) const requiredVyanjans = currentWordData.parts.filter(p => !matraList.includes(p)); const fullVyanjanPool = [...new Set([...requiredVyanjans, ...currentWordData.extraVyanjan])]; // 3. Create the Matra Pool (Required + Extras) const requiredMatras = currentWordData.parts.filter(p => matraList.includes(p)); const fullMatraPool = [...new Set([...requiredMatras, ...currentWordData.extraMatra])]; // 4. Populate Vyanjan Mixer (Shuffled) fullVyanjanPool.sort(() => Math.random() - 0.5).forEach(char => { createTile(char, 'vyanjan', vMixer); }); // 5. Populate Matra Mixer (Shuffled) fullMatraPool.sort(() => Math.random() - 0.5).forEach(char => { createTile(char, 'matra', mMixer); }); // ... rest of your tray generation code ... }
